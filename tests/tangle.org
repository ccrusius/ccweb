* 001: Basic Haskell indentation

#+BEGIN_SRC haskell :tangle "tests/out/tangle001.hs"
main :: IO ()
main = do
  <<main>>

hello :: String
hello = "Hello, world!"
#+END_SRC

#+NAME: main
#+BEGIN_SRC haskell
putStrLn str
where
  <<str>>
#+END_SRC

#+NAME: str
#+BEGIN_SRC haskell
str = hello
#+END_SRC

* 004: Line number pragmas
:PROPERTIES:
:header-args: :tangle "tests/out/tangle004.hs" :comments yes
:END:

#+BEGIN_SRC haskell
main :: IO ()
main = do
  putStrLn str
  <<Put string again>>
  putStrLn str
#+END_SRC

#+BEGIN_SRC haskell
str :: String
str = "Hello, world!"
#+END_SRC

#+NAME: Put string again
#+BEGIN_SRC haskell
putStrLn str
#+END_SRC
