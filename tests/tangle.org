  <<main>>

  <<str>>

* 004: Line number pragmas
:PROPERTIES:
:header-args: :tangle "tests/out/tangle004.hs"
:END:

#+BEGIN_SRC haskell :comments yes
main :: IO ()
main = do
  putStrLn str
  <<Put string again>>
  putStrLn "(" >> <<Put string again>> >> putStrLn ")"
  putStrLn str
#+END_SRC

#+BEGIN_SRC haskell :comments yes
str :: String
str = "Hello, world!"
<<Inherited custom comments>>
<<Inherited standard comments>>
<<Inherited no comments>>
#+END_SRC

#+NAME: Put string again
#+BEGIN_SRC c++
// The line pragma should still be a Haskell one
std::cerr << "This is not C++" << std::endl;
#+END_SRC

#+NAME: Inherited custom comments
#+BEGIN_SRC haskell :comments "--!! "
-- Line pragma above should start with --!!
<<Sub-section with inherited comment style>>
#+END_SRC

#+NAME: Sub-section with inherited comment style
#+BEGIN_SRC haskell
-- And the pragma above should start with --!! too.
#+END_SRC

#+NAME: Inherited standard comments
#+BEGIN_SRC haskell
-- There should be a regular line pragma above
#+END_SRC

#+NAME: Inherited no comments
#+BEGIN_SRC haskell :comments no
-- There should be no regular line pragma above pointing
-- to this source block.
#+END_SRC
