  <<str>>

* 004: Line number pragmas and sections
:PROPERTIES:
:header-args: :tangle "tests/out/tangle004.hs" :comments yes
:END:

#+BEGIN_SRC haskell
main :: IO ()
main = do
  putStrLn str
  <<Put string again>>
  putStrLn "(" >> <<Put string again>> >> putStrLn ")"
  putStrLn str
#+END_SRC

#+BEGIN_SRC haskell
str :: String
str = "Hello, world!"
<<Custom comments>>
#+END_SRC

#+NAME: Put string again
#+BEGIN_SRC c++
// The line pragma should still be a Haskell one
std::cerr << "This is not C++" << std::endl;
#+END_SRC
